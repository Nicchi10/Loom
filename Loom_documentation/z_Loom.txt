@startuml
package Core {

    package Enums {
        enum MessageRole {
            System
            User
            Assistant
            Tool
        }

        enum MemoryMode {
            None
            FullHistory
            Summary
            Extractive
            Hybrid
        }

        enum InjectionStrategy {
            Sectioned
            SystemXML
            SystemJSON
        }

        enum ExecutionPriority {
            Balanced
            LatencyOptimized
            CostOptimized
            QualityOptimized
        }
    }

    package Interfaces {

        interface ILlmInvocation {
            + Conversation : ConversationState
            + Rag : RagContext
            + Memory : MemoryContext
            + Tools : ToolContext
            + Options : GenerationOptions
            + Hints : ExecutionHints

            + Validate() : IValidationResult
        }

        interface IProviderAdapter {
            + ProviderName : String

            + ExecuteAsync(ILlmInvocation) : Task<LlmResponse>
            + SupportsCapability(String) : Boolean
        }

        interface IValidationResult {
            + IsValid : Boolean
            + Errors : IEnumerable<String>
        }

        interface ITool {
            + ToolName : String
            + Description : String
            + Parameters : List<ToolParameters>
            + Required : List<String>
            + Strictness : Boolean?

            + ExecuteAsync(rawInput: Dictionary<String, Object>) : Task<String>
        }
    }

    package Models {

        class LlmInvocation {
            + Conversation : ConversationState
            + Rag : RagContext
            + Memory : MemoryContext
            + Tools : ToolContext
            + Options : GenerationOptions
            + Hints : ExecutionHints

            + Validate() : IValidationResult
        }

        ILlmInvocation <|.. LlmInvocation

        class Message {
            + Role : MessageRole
            + Content : String
            + ToolCallId : String
            + ToolName : String
            + ToolArgs : Dictionary<String, Object>
            + Metadata : Dictionary<String, Object>
        }

        class ConversationState {
            + TraceId : String
            + SystemPrompt : String
            + TokenBudget : Int?
            + Messages : List<Message>
            + TurnIndex : Int = 0
        }

        class ExecutionHints {
            + PreferredModel : String
            + Priority : ExecutionPriority
            + MaxRetryCount : Int = 3
        }

        class GenerationOptions {
            + Temperature : Float?
            + TopK : Float?
            + MaxTokens : Int?
            + StopSequences : List<String>
            + JsonMode : Boolean = False
        }

        class MemoryContext {
            + Mode : MemoryMode
            + Content : String
            + TTL : TimeSpan?
        }

        class ModelMetaData {
            + ModelId : String
            + ProviderName : String
            + ContextWindow : Int
            + SupportsTools : Boolean
            + CostLevel : Int
            + Capabilities : List<String>
        }

        class RagChunk {
            + SourceId : String
            + Score : Double
            + Text : String
            + Metadata : Dictionary<String, Object>
        }

        class RagContext {
            + Strategy : InjectionStrategy
            + MaxTokens : Int = 2000
            + RetrievedChunks : List<RagChunk>
        }

        class ToolDefinition {
            + ToolName : String
            + Description : String
            + Parameters : List<ToolParameters>
            + Required : List<String>
            + Strictness : Boolean?
        }

        class ToolParameters {
            + Name : String
            + Type : String
            + Description : String
            + Enums : List<String>
        }

        class ToolCallInvocation {
            + Type : String = "function_call"
            + CallId : String
            + ToolName : String
            + Arguments : Dictionary<String, Object>
        }

        class ToolContext {
            + RegisteredTools : List<ToolDefinition>
            + MaxToolDepth : Int = 5
        }

        class LlmResponse {
            + Id : String
            + ModelUsed : String
            + Role : MessageRole
            + Content : String
            + ToolCalls : List<ToolCallInvocation>
            + PromptTokens : Integer
            + CompletionTokens : Integer
            + TotalTokens : Integer
            + Metadata : Dictionary<String, Object>
        }

        class ValidationResult {
            - _isValid : Boolean
            - _errors : IEnumerable<String>

            + IsValid : Boolean
            + Errors : IEnumerable<String>

            + Success() : ValidationResult
            + Failure(errors : IEnumerable<String>) : ValidationResult
        }

        IValidationResult <|.. ValidationResult
    }

    class InvocationValidator {
        + Validate(ILlmInvocation) : ValidationResult
    }

}
@enduml

@startuml
package Engine {

    package Assemblers {
        class PromptAssembler {
            + Assemble(LlmInvocation) : List<Message>
            + AssembleInputItems(LlmInvocation) : List<AssembledItem>
            + BuildContextBlock(LlmInvocation) : String
            + AddContent(String, String) : String
        }
    }

    package Context {

        class ToolRegistry {
            - _invocation : LlmInvocation
            - _executors : Dictionary<String, ITool>
            + New(LlmInvocation)
            + RegisterTool(ITool)
            + ExecuteAsync(String, Dictionary<String,Object>) : Task<String>
        }

        class RagManager {
            - _invocation : LlmInvocation
            + New(LlmInvocation)
            + HasContext : Boolean
            + AddResults(IEnumerable<RagChunk>, Boolean?)
            + SetStrategy(InjectionStrategy)
        }

        class MemoryManager {
            - _invocation : LlmInvocation
            + New(LlmInvocation)
            + UpdateMemory(String?)
            + IsMemoryExpired(DateTime) : Boolean
        }

        class ConversationManager {
            - _invocation : LlmInvocation
            + AddMessage(MessageRole, String, Dictionary<String, Object>?)
            + AddUserRequest(String)
            + AddToolResult(String, String, String, Dictionary<String, Object>, Dictionary<String, Object>?)
            + ResetHistory(Boolean)
        }
    }

    package Infrastructure {

        class TokenCounter {
            + CountTokens(String) : Int
            + CalculateTotal(LlmInvocation) : Int
            + IsWithinBudget(LlmInvocation) : Boolean
        }

        class ExecutionRouter {
            - _adapters : List<IProviderAdapter>
            - _catalog : ModelCatalog
            + RegisterAdapter(IProviderAdapter)
            + Route(ILlmInvocation) : IProviderAdapter
        }

        class ModelCatalog {
            - _models : List<ModelMetaData>
            + GetModelInfo(String) : ModelMetaData
            + GetFallBack(ExecutionPriority) : ModelMetaData
        }
    }

    class LoomClient {
        + Invocation : LlmInvocation
        + Conversation : ConversationManager
        + Rag : RagManager
        + Memory : MemoryManager
        + Tool : ToolRegistry
        - _router : ExecutionRouter
        - _tokenCounter : TokenCounter
        + New()
        + RegisterProvider(IProviderAdapter)
        + SendAsync() : Task<LlmResponse>
    }

}
@enduml
